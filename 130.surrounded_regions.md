            """

            Oarea = []
            s = []
            s.append((i,j))
            while s:
                (x,y) = s.pop()
                if x == 0 or x == row - 1 or y == 0 or y == col -1 :
                    shouldChange = False
                visited[x][y] = 1
                Oarea.append((x,y))
                if legal(x-1,y):
                    s.append((x-1,y))

                return Oarea,shouldChange

        def legal(x,y):
